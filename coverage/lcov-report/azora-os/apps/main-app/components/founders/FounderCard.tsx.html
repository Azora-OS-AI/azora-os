import React from 'react';
import { motion } from 'framer-motion';
import { Mail, Phone, Award, Calendar, Shield, Bot, LucideIcon } from 'lucide-react';
import { Founder } from '../../features/founders/mockData';
 
const getStatusBadge = (status: Founder['status']) => {
  const styles = {
    active: "bg-green-500/10 text-green-300 border-green-500/30",
    inactive: "bg-gray-500/10 text-gray-400 border-gray-500/30",
    'on-leave': "bg-yellow-500/10 text-yellow-300 border-yellow-500/30",
  };
  return <span className={`px-3 py-1 rounded-full text-xs font-semibold border ${styles[status]}`}>{status.replace('-', ' ').toUpperCase()}</span>;
};
 
const InfoItem: React.FC<{icon: LucideIcon, text: string, href?: string}> = ({icon: Icon, text, href}) => {
  const content = href ? <a href={href} className="hover:underline text-cyan-300">{text}</a> : <span>{text}</span>;
  return (
    <div className="flex items-center space-x-3 text-gray-400">
      <Icon className="w-5 h-5 text-cyan-400 flex-shrink-0" />
      {content}
    </div>
  );
}
 
const FounderCard: React.FC<{ founder: Founder }> = ({ founder }) => {
  const isAI = founder.roles.includes('AI_DEPUTY_CEO');
  
  return (
    <motion.div
      variants={{ hidden: { opacity: 0, y: 20 }, visible: { opacity: 1, y: 0 } }}
      className="bg-gray-950/70 border border-cyan-500/20 rounded-2xl overflow-hidden shadow-2xl shadow-cyan-500/5 backdrop-blur-lg transition-all duration-300 hover:border-cyan-500/40 hover:shadow-cyan-500/10"
    >
      <div className={`p-6 ${isAI ? 'bg-gradient-to-br from-cyan-900/50 to-purple-900/30' : 'bg-gradient-to-br from-gray-800/50 to-gray-900/30'}`}>
        <div className="flex items-start justify-between mb-4">
          <div className="flex items-center space-x-4">
            <div className={`w-20 h-20 rounded-full flex items-center justify-center text-3xl font-bold border-2 ${isAI ? 'bg-cyan-500/20 text-cyan-300 border-cyan-400/50' : 'bg-gray-700/50 border-gray-600'}`}>
              {isAI ? <Bot size={40}/> : founder.name.split(' ').map(n => n[0]).join('')}
            </div>
            <div>
              <h2 className="text-2xl font-bold text-white">{founder.name}</h2>
              <div className="flex flex-wrap gap-2 mt-2">
                {founder.titles.map((title, idx) => (
                  <span key={idx} className="text-xs bg-cyan-500/10 px-2 py-1 rounded text-cyan-200 border border-cyan-500/20">
                    {title}
                  </span>
                ))}
              </div>
            </div>
          </div>
          {getStatusBadge(founder.status)}
        </div>
        
        <div className="flex items-center space-x-4 text-sm text-gray-300">
          <div className="flex items-center space-x-2 bg-gray-800/50 px-3 py-1.5 rounded-md border border-gray-700">
            <Award className="w-4 h-4 text-yellow-400" />
            <span>Equity: <span className="font-bold text-white">{founder.equity}%</span></span>
          </div>
          {founder.votingRights && (
            <div className="flex items-center space-x-2 bg-gray-800/50 px-3 py-1.5 rounded-md border border-gray-700">
              <Shield className="w-4 h-4 text-green-400" />
              <span className="text-white">Voting Rights</span>
            </div>
          )}
        </div>
      </div>
 
      <div className="p-6 space-y-5">
        <p className="text-sm text-gray-300/90 leading-relaxed italic">
          "{founder.bio}"
        </p>
 
        <div className="border-t border-cyan-500/20 pt-4 space-y-2">
          <InfoItem icon={Mail} text={founder.email} href={`mailto:${founder.email}`} />
          {founder.phoneNumber && <InfoItem icon={Phone} text={founder.phoneNumber} />}
          <InfoItem icon={Calendar} text={`Joined: ${new Date(founder.joinedDate).toLocaleDateString('en-US', { year: 'numeric', month: 'long', day: 'numeric' })}`} />
        </div>
 
        <div className="border-t border-cyan-500/20 pt-4">
          <h3 className="font-semibold text-cyan-100 mb-3">Key Responsibilities</h3>
          <div className="flex flex-wrap gap-2">
            {founder.permissions.slice(0, 6).map((perm, idx) => (
              <span key={idx} className={`text-xs px-2.5 py-1 rounded-full border ${isAI ? 'bg-cyan-900/50 text-cyan-300 border-cyan-800/50' : 'bg-purple-900/50 text-purple-300 border-purple-800/50'}`}>
                {perm.replace(/_/g, ' ').toLowerCase()}
              </span>
            ))}
            {founder.permissions.length > 6 && (
              <span className="text-xs px-2.5 py-1 bg-gray-800/50 text-gray-400 rounded-full border border-gray-700/50">
                +{founder.permissions.length - 6} more
              </span>
            )}
          </div>
        </div>
        
        <div className="border-t border-cyan-500/20 pt-4 text-right">
          <p className="text-xs text-gray-500">
            Founder ID: <span className="font-mono font-semibold text-gray-400">{founder.founderId}</span>
          </p>
        </div>
      </div>
    </motion.div>
  );
};
 
export default FounderCard;
